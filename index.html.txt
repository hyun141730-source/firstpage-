  <script>
    // 글자 크기 조절 및 고대비 토글 (간단한 접근성 기능)
    const root = document.documentElement;
    const increase = document.getElementById('textIncrease');
    const decrease = document.getElementById('textDecrease');
    const contrast = document.getElementById('highContrast');

    increase.addEventListener('click', ()=>{
      const cur = parseFloat(getComputedStyle(document.body).fontSize);
      document.body.style.fontSize = (cur + 2) + 'px';
    });
    decrease.addEventListener('click', ()=>{
      const cur = parseFloat(getComputedStyle(document.body).fontSize);
      if(cur>12) document.body.style.fontSize = (cur - 2) + 'px';
    });
    contrast.addEventListener('click', ()=>{
      const pressed = contrast.getAttribute('aria-pressed') === 'true';
      contrast.setAttribute('aria-pressed', String(!pressed));
      if(!pressed){
        document.body.style.background = '#000';
        document.body.style.color = '#fff';
      } else {
        document.body.style.background = '';
        document.body.style.color = '';
      }
    });

    // 키보드 포커스 스타일 보강
    document.addEventListener('keydown', (e)=>{
      if(e.key === 'Tab') document.documentElement.classList.add('show-focus');
    });
  </script>
